//master writer
#include <Wire.h>
int x=0;
int led=5;
void setup() {
  // put your setup code here, to run once:
  Wire.begin();
  pinMode(led, OUTPUT);
}

void loop() {
  // put your main code here, to run repeatedly:
  digitalWrite(led, HIGH);
  delay(1000);
  Wire.beginTransmission(1);
  Wire.write(x);
  Wire.endTransmission();
  digitalWrite(led, LOW);
  delay(1000);
  x++;
  if(x>5) 
  {
    x=0;
  }
}
